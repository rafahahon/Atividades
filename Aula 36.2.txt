#include <LiquidCrystal_I2C.h>

/*
Autor: Rafaella
Nome programa: Circuito com botao up e botao down
Descricao: botao com funcao up faz a seta ir pra cima e o
down para baixo
Data: 13/03/25
*/


LiquidCrystal_I2C lcd(0x27, 16, 2);
#define pinledWhite 8
#define pinledBlue 7

void setup()
{
  lcd.init();
  lcd.backlight();
  Serial.begin(9600);
  pinMode(pinledWhite, OUTPUT);
  pinMode(pinledBlue, OUTPUT);
}

void loop()
{
  float leitura = analogRead(A0);
  lcd.setCursor(1, 0);
  lcd.print("Led Azul");
  lcd.setCursor(1, 1);
  lcd.print("Led Branco");
  
    if(leitura == 0){
    lcd.setCursor(0, 0);
    lcd.print(">");
    lcd.setCursor(0, 1);
    lcd.print(" ");
  } else if(leitura >= 130.5 && leitura <= 159.5){
    lcd.setCursor(0, 1);
    lcd.print(">");
    lcd.setCursor(0, 0);
    lcd.print(" ");
  }
}
